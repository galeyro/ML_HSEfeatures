    return accuracy_score(Ytest, Ypred)

def get_metrics(y_true, y_pred):
    cm = confusion_matrix(y_true, y_pred, labels=[0, 1])
    tn, fp, fn, tp = cm.ravel()
    acc = (tp + tn) / (tp + tn + fp + fn)
    tnr = tn / (tn + fp) if (tn + fp) > 0 else 0  # Specificidad
    return acc, tnr, cm
